pipeline {
  agent any

  tools {
    nodejs "NodeJS 18"  // make sure this matches your Jenkins NodeJS tool config
  }

  environment {
    FRONTEND_DIR = 'frontend'
    BACKEND_DIR = 'backend'
  }

  stages {
    stage('Checkout') {
      steps {
        git 'https://github.com/your-username/your-repo.git'
      }
    }

    stage('Install Dependencies') {
      steps {
        dir("$FRONTEND_DIR") {
          sh 'npm install'
        }
        dir("$BACKEND_DIR") {
          sh 'npm install'
        }
      }
    }

    stage('Build Frontend') {
      steps {
        dir("$FRONTEND_DIR") {
          sh 'npm run build'
        }
      }
    }

    stage('Test Backend') {
      steps {
        dir("$BACKEND_DIR") {
          sh 'npm test || echo "Tests failed or not present"'
        }
      }
    }

    stage('Deploy (optional)') {
      steps {
        echo "Add your deployment script here"
      }
    }
  }
}
